import { Component } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Router } from '@angular/router';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css'] 
})
/**
 * LoginComponent é uma classe que representa o componente de login da aplicação.
 * Ele lida com a autenticação de usuários e o redirecionamento para a página inicial.
 */
export class LoginComponent {
  /** O endereço de e-mail inserido pelo usuário. */
  email: string = "";

   /** A senha inserida pelo usuário. */
  senha: string = "";

  /** URL da API utilizada para autenticação de usuários. */
  readonly apiURL = 'https://64e6a1f309e64530d1801dd3.mockapi.io/api/v1/Nome/';

  /**
   * Construtor da classe LoginComponent.
   * @param http Um serviço HTTP para fazer requisições à API.
   * @param router Um serviço de roteamento para redirecionar os usuários após o login bem-sucedido.
   */
  constructor(private http: HttpClient, private router: Router){} 

  /**
   * Método ngOnInit é chamado após a inicialização do componente.
   * Neste componente, não há nenhuma inicialização específica a ser realizada, então o método está vazio.
   */
  ngOnInit(): void{}
  
  
  /**
   * Método para verificar as credenciais do usuário e autenticá-lo.
   */
  verificarUsuario(){
    window.alert('Verificando dados do usuário...');
    var variavel;
    console.log('Email:', this.email);
    console.log('Senha:', this.senha);

    // Verifica a senha com a expressão regular
    const regex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9!@#\$%\^&\*\(\)_\+\-=\[\]\{\};:'",<>\./?\\|]).{6,}$/;
    if (!regex.test(this.senha)) {
      window.alert('Senha fora dos padrões');
      return;
    }

    // URL do servidor
    const url = `https://64e6a1f309e64530d1801dd3.mockapi.io/api/v1/Nome?email=${this.email}` 

    variavel = this.http.get<usuario[]>(url).subscribe(
      (response: usuario[]) => {
        console.log(response);
    
        // Encontrar o usuário com o email fornecido na resposta
        const user = response.find(user => user.email === this.email);
    
        if (user) {
          // Usuário encontrado, agora compare as senhas
          if (user.senha === this.senha) {
            window.alert('Login efetuado com sucesso');
            this.router.navigate(['./pag-inicial']);
          } else {
            window.alert('Email ou senha Incorretos');
          }
        } else {
          // Usuário não encontrado
          window.alert('Usuário não encontrado');
        }
      },
      error => {
        console.error(error);
        window.alert('Ocorreu um erro ao efetuar o login');
      }
    );
  }

  /**
 * Método `cadastra()` da classe `LoginComponent`.
 * Este método é utilizado para cadastrar um novo usuário no sistema.
 * Ele faz uma requisição POST para a API com as informações de e-mail e senha do usuário.
 */
  cadastra(){

    // Exibe os dados de e-mail e senha para teste
    alert(this.senha)
    alert(this.email)

    // Cria um objeto com as informações do aluno (usuário)
    var aluno = {
      email: this.email,
      senha: this.senha
    }

    // Faz uma requisição POST para a API com as informações do aluno
   this.http.post(this.apiURL, aluno)
   .subscribe(
    resultado => {
      console.log(resultado)
      // Lógica para lidar com o resultado bem-sucedido, se necessário
    },
    erro => {
      if(erro.status == 400){
        console.log(erro)
        // Lógica para lidar com erros de status 400, se necessário
      }
    }
   )

  }
}
